  set(CHAI3D_VERSION_FILE "${PROJECT_SOURCE_DIR}/version")

file(READ "${CHAI3D_VERSION_FILE}" CHAI3D_VERSION_CONTENT)

foreach(version_component MAJOR MINOR)
  string(REGEX MATCH "${version_component}=([0-9]+)" _version_match
         "${CHAI3D_VERSION_CONTENT}")
  if(NOT CMAKE_MATCH_1)
    message(FATAL_ERROR
            "Unable to parse ${version_component} from ${CHAI3D_VERSION_FILE}")
  endif()
  set(${version_component}_VERSION "${CMAKE_MATCH_1}")
endforeach()

set(PATCH_VERSION "0")
if(CHAI3D_VERSION_CONTENT MATCHES "PATCH=([0-9]+)")
  set(PATCH_VERSION "${CMAKE_MATCH_1}")
elseif(CHAI3D_VERSION_CONTENT MATCHES "RELEASE=([0-9]+)")
  set(PATCH_VERSION "${CMAKE_MATCH_1}")
endif()

set(PRERELEASE_VERSION "")
if(CHAI3D_VERSION_CONTENT MATCHES "PRERELEASE=([0-9]+)")
  set(PRERELEASE_VERSION "${CMAKE_MATCH_1}")
endif()

set(PROJECT_VERSION
    "${MAJOR_VERSION}.${MINOR_VERSION}.${PATCH_VERSION}${PRERELEASE_VERSION}")

  set(OS "win")

    set(ARCH "x64")

  endif()
elseif(${CMAKE_SYSTEM_NAME} MATCHES Linux)
  set(OS "lin")
  if(CMAKE_SYSTEM_PROCESSOR MATCHES "^(x86_64|amd64)$")
    set(ARCH "x86_64")
  elseif(CMAKE_SYSTEM_PROCESSOR MATCHES "^i[3-6]86$")
    set(ARCH "i686")
  else()
    set(ARCH "${CMAKE_SYSTEM_PROCESSOR}")
  endif()
elseif(${CMAKE_SYSTEM_NAME} MATCHES Darwin)
  set(OS "mac")
  if(CMAKE_HOST_SYSTEM_PROCESSOR)
    set(ARCH "${CMAKE_HOST_SYSTEM_PROCESSOR}")
  else()
    execute_process(
      COMMAND uname -m
      COMMAND tr -d "\n"
      OUTPUT_VARIABLE ARCH)
  endif()
endif()

function(chai3d_collect_sources output_variable relative_dir)
  if(EXISTS "${CHAI3D_VENDOR_ROOT}/${relative_dir}")
    file(GLOB_RECURSE collected_sources RELATIVE ${PROJECT_SOURCE_DIR}
         ${CHAI3D_VENDOR_ROOT}/${relative_dir}/*.c
         ${CHAI3D_VENDOR_ROOT}/${relative_dir}/*.cpp
         ${CHAI3D_VENDOR_ROOT}/${relative_dir}/*.h)
    set(${output_variable} ${collected_sources} PARENT_SCOPE)
  endif()
endfunction()

foreach(group_file
        ${source}
        ${CHAI3D_EIGEN_SOURCES}
        ${CHAI3D_GIF_SOURCES}
        ${CHAI3D_GLEW_SOURCES}
        ${CHAI3D_3DS_SOURCES}

        ${CHAI3D_OPENAL_SOURCES}
        ${CHAI3D_THEORA_SOURCES}
        ${CHAI3D_RTMIDI_SOURCES})
  get_filename_component(PARENT_DIR "${group_file}" PATH)
  string(REGEX REPLACE "(\\./)?(src|include)/?" "" GROUP "${PARENT_DIR}")
  string(REPLACE "/" "\\" GROUP "${GROUP}")
  source_group("${GROUP}" FILES "${group_file}")
endforeach()
