cmake_minimum_required(VERSION 3.10)

project(SAI-COMMON)
set(PROJECT_VERSION 0.1.0)

option(BUILD_EXAMPLES "Build examples" ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
  message(STATUS "Setting build type to 'Release' as none was specified.")
  set(CMAKE_BUILD_TYPE
      Release
      CACHE STRING "Choose the type of build." FORCE)
  set_property(CACHE CMAKE_BUILD_TYPE PROPERTY STRINGS "Debug" "Release"
                                               "MinSizeRel" "RelWithDebInfo")
endif()

find_package(Eigen3 REQUIRED)

set(SAI_COMMON_EIGEN_INCLUDE_DIRS "")
if(TARGET Eigen3::Eigen)
  get_target_property(SAI_COMMON_EIGEN_INCLUDE_DIRS Eigen3::Eigen
                      INTERFACE_INCLUDE_DIRECTORIES)
endif()
if(NOT SAI_COMMON_EIGEN_INCLUDE_DIRS)
  if(DEFINED EIGEN3_INCLUDE_DIRS AND NOT "${EIGEN3_INCLUDE_DIRS}" STREQUAL "")
    set(SAI_COMMON_EIGEN_INCLUDE_DIRS ${EIGEN3_INCLUDE_DIRS})
  else()
    set(SAI_COMMON_EIGEN_INCLUDE_DIRS ${EIGEN3_INCLUDE_DIR})
  endif()
endif()
if(DEFINED ENV{CONDA_PREFIX} AND EXISTS "$ENV{CONDA_PREFIX}/include/eigen3")
  list(APPEND SAI_COMMON_EIGEN_INCLUDE_DIRS "$ENV{CONDA_PREFIX}/include/eigen3")
endif()
list(REMOVE_DUPLICATES SAI_COMMON_EIGEN_INCLUDE_DIRS)

find_package(jsoncpp CONFIG QUIET)
if(TARGET jsoncpp_lib)
  set(JSONCPP_LIBRARY jsoncpp_lib)
elseif(TARGET JsonCpp::JsonCpp)
  set(JSONCPP_LIBRARY JsonCpp::JsonCpp)
elseif(CMAKE_SYSTEM_NAME MATCHES Linux)
  find_package(PkgConfig REQUIRED)
  pkg_check_modules(JSONCPP jsoncpp)
  link_directories(${JSONCPP_LIBRARY_DIRS})
  set(JSONCPP_LIBRARY ${JSONCPP_LIBRARIES})
else()
  find_library(JSONCPP_LIBRARY jsoncpp REQUIRED)
endif()

find_library(HIREDIS_LIBRARY hiredis REQUIRED)
find_path(HIREDIS_INCLUDE_DIR hiredis/hiredis.h REQUIRED)

set(REDIS_SOURCE ${PROJECT_SOURCE_DIR}/src/redis/RedisClient.cpp)
set(TIMER_SOURCE ${PROJECT_SOURCE_DIR}/src/timer/LoopTimer.cpp)
set(FILTER_SOURCE ${PROJECT_SOURCE_DIR}/src/filters/ButterworthFilter.cpp)
set(LOGGER_SOURCE ${PROJECT_SOURCE_DIR}/src/logger/Logger.cpp)

include_directories(${PROJECT_SOURCE_DIR}/src ${SAI_COMMON_EIGEN_INCLUDE_DIRS}
                    ${JSONCPP_INCLUDE_DIRS} ${HIREDIS_INCLUDE_DIR})

add_library(sai-common STATIC ${REDIS_SOURCE} ${TIMER_SOURCE} ${FILTER_SOURCE}
                               ${LOGGER_SOURCE})
target_link_libraries(sai-common PUBLIC ${JSONCPP_LIBRARY} ${HIREDIS_LIBRARY})

set(SAI-COMMON_LIBRARIES sai-common ${JSONCPP_LIBRARY} ${HIREDIS_LIBRARY})
set(SAI-COMMON_INCLUDE_DIRS ${PROJECT_SOURCE_DIR}/src)

export(TARGETS sai-common FILE ${PROJECT_BINARY_DIR}/SAI-COMMONTargets.cmake)

set(CONF_INCLUDE_DIRS ${SAI-COMMON_INCLUDE_DIRS} ${JSONCPP_INCLUDE_DIRS}
                      ${HIREDIS_INCLUDE_DIR})
set(CONF_LIBRARIES ${SAI-COMMON_LIBRARIES})

configure_file(SAI-COMMONConfig.cmake.in
               "${PROJECT_BINARY_DIR}/SAI-COMMONConfig.cmake" @ONLY)
include(CMakePackageConfigHelpers)
write_basic_package_version_file(
  "${CMAKE_CURRENT_BINARY_DIR}/SAI-COMMONConfigVersion.cmake"
  VERSION ${PROJECT_VERSION}
  COMPATIBILITY SameMajorVersion)

if(BUILD_EXAMPLES)
  add_subdirectory(${PROJECT_SOURCE_DIR}/examples)
endif()
